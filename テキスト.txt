function setStatus(t,m){statusEl.className='status'+(t?' '+t:'');statusEl.textContent=m||'';}

async function geocode(text){
  const url=`https://maps.googleapis.com/maps/api/geocode/json?address=${encodeURIComponent(text)}&region=jp&language=ja&key=${GOOGLE_KEY}`;
  const res=await fetch(url);
  if(!res.ok)throw new Error(`場所検索でエラー(${res.status})が発生しました。`);
  const data=await res.json();
  if(data.status==='REQUEST_DENIED')throw new Error('APIキーが無効か、制限設定が合っていません。'+(data.error_message||''));
  if(data.status==='OVER_QUERY_LIMIT')throw new Error('利用上限に達しました。');
  if(data.status!=='OK'||!data.results.length)throw new Error(`「${text}」が見つかりませんでした。市区町村名を付けて試してください。`);
  const loc=data.results[0].geometry.location;
  return [loc.lng,loc.lat]; // [経度,緯度] で従来コードと互換
}
async function legKm(a,b){
  // a,b は [経度,緯度]。Google Directionsは 緯度,経度 の順で渡す
  const origin=`${a[1]},${a[0]}`,dest=`${b[1]},${b[0]}`;
  const url=`https://maps.googleapis.com/maps/api/directions/json?origin=${origin}&destination=${dest}&mode=driving&language=ja&key=${GOOGLE_KEY}`;
  const res=await fetch(url);
  if(!res.ok)throw new Error(`距離計算でエラー(${res.status})が発生しました。`);
  const data=await res.json();
  if(data.status==='REQUEST_DENIED')throw new Error('APIキーが無効か、制限設定が合っていません。'+(data.error_message||''));
  if(data.status==='OVER_QUERY_LIMIT')throw new Error('利用上限に達しました。');
  if(data.status!=='OK'||!data.routes.length)throw new Error('ルートが見つかりませんでした。');
  // 全leg合算(経由なし単区間なので通常1leg)
  const meters=data.routes[0].legs.reduce((s,l)=>s+l.distance.value,0);
  return meters/1000;
}
// 緯度経度から直線距離(km)。geocodeミス検出用
function straightKm(a,b){
  const R=6371,toR=d=>d*Math.PI/180;
  const dLat=toR(b[1]-a[1]),dLon=toR(b[0]-a[0]);
  const s=Math.sin(dLat/2)**2+Math.cos(toR(a[1]))*Math.cos(toR(b[1]))*Math.sin(dLon/2)**2;
  return R*2*Math.asin(Math.sqrt(s));
}

function modeLabel(m){return m==='round'?'往復':m==='loop'?'周回':'片道';}

function render(){
  entriesEl.innerHTML='';
  if(!entries.length){entriesEl.innerHTML='<tr class="empty"><td colspan="5">まだ区間が追加されていません</td></tr>';exportBtn.disabled=true;}
  else{
    exportBtn.disabled=false;
    entries.forEach((e,i)=>{
      const tr=document.createElement('tr');
      const routeText=e.points.join(' → ')+(e.mode==='loop'?' → '+e.points[0]:'');
      tr.innerHTML=`<td>${e.date.slice(5).replace('-','/')}</td>`+
        `<td class="t-route">${routeText}<span class="tag">${modeLabel(e.mode)}</span><span class="sub">${e.totalKm.toFixed(1)}km ／ ${e.rate}円</span></td>`+
        `<td>${e.totalKm.toFixed(1)}km</td><td class="t-yen">¥${e.amount.toLocaleString()}</td>`+
        `<td class="t-del"><button data-i="${i}" aria-label="削除">×</button></td>`;
      entriesEl.appendChild(tr);
    });
  }
  const total=entries.reduce((s,e)=>s+e.amount,0);
  totalAmountEl.textContent=`¥${total.toLocaleString()}`;
  totalCell.textContent=`¥${total.toLocaleString()}`;
}

calcBtn.addEventListener('click',async()=>{
  const names=[...ptList.querySelectorAll('.pt-item input')].map(i=>i.value.trim());
  if(names.some(n=>!n)){setStatus('error','すべての地点を入力してください。');return;}
  if(names.length<2){setStatus('error','出発地と到着地が必要です。');return;}
  const rate=Number(rateInput.value)||0,date=dateInput.value;
  calcBtn.disabled=true;setStatus('loading','住所を検索して距離を計算しています…');
  try{
    const coords=[];
    for(const n of names){coords.push(await geocode(n));}
    // geocodeミス検出:隣接地点が直線で500km以上離れていたら警告
    for(let i=0;i<coords.length-1;i++){
      if(straightKm(coords[i],coords[i+1])>500){
        setStatus('error',`「${names[i]}」か「${names[i+1]}」の場所がうまく特定できていない可能性があります。市区町村名を付けるなど、もう少し詳しい地名で試してください。`);
        calcBtn.disabled=false;return;
      }
    }
    // 各区間を順に計算
    let oneWay=0;
    for(let i=0;i<coords.length-1;i++){oneWay+=await legKm(coords[i],coords[i+1]);}
    let totalKm=oneWay;
    if(mode==='round'){totalKm=oneWay*2;}
    else if(mode==='loop'){totalKm=oneWay+await legKm(coords[coords.length-1],coords[0]);}
    const amount=Math.round(totalKm*rate);
    entries.push({date,points:[...names],mode,totalKm,rate,amount});
    render();setStatus('','');
    // 入力欄リセット(地点は2個に戻す)
    ptList.innerHTML='';addPoint();addPoint();
  }catch(err){setStatus('error',err.message||'通信エラーが発生しました。電波状況を確認して再度お試しください。');}
  finally{calcBtn.disabled=false;}
});

  const ws=XLSX.utils.aoa_to_sheet(rows);

  const wb=XLSX.utils.book_new();
  XLSX.utils.book_append_sheet(wb,ws,'交通費');
  XLSX.writeFile(wb,`交通費請求書_${new Date().toISOString().slice(0,10)}.xlsx`);
});
